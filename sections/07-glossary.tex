\section*{Appendix: Glossary of Acronyms}
\addcontentsline{toc}{section}{Appendix: Glossary of Acronyms}

\begin{note}
The following acronyms and abbreviations are used throughout this project plan. Definitions are provided at first occurrence in the text; this glossary serves as a reference guide.
\end{note}

\begin{itemize}
    \item \textbf{AIFS ENS} --- Artificial Intelligence Foundation Series Ensemble (ECMWF)
    \item \textbf{AAU} --- Aalborg University
    \item \textbf{CaMa Flood} --- Catchment Modelling and Analysis (routing model)
    \item \textbf{CAMELS DK} --- Catchment Attributes and MEteoroLogy for Large-sample Studies, Denmark
    \item \textbf{CDF} --- Cumulative Distribution Function (matching technique)
    \item \textbf{CDI} --- Copernicus Drought Index
    \item \textbf{CRPS} --- Continuous Ranked Probability Score
    \item \textbf{CSE-UA} --- Shuffled Complex Evolution University of Arizona (optimisation algorithm)
    \item \textbf{DFF} --- Danish Independent Research Fund
    \item \textbf{DK model} --- Danish hydrological model
    \item \textbf{DMI} --- Danish Meteorological Institute
    \item \textbf{dPL} --- Differentiable Parameter Learning
    \item \textbf{DREAM} --- Drought Resilience through satellite-integrated Early warning And Monitoring
    \item \textbf{EDgE} --- ECMWF Drought Ensemble Global forecasting
    \item \textbf{ECMWF} --- European Centre for Medium-Range Weather Forecasts
    \item \textbf{ECMWF IFS} --- ECMWF Integrated Forecasting System
    \item \textbf{EDO CDI v4} --- European Drought Observatory Copernicus Drought Index version 4
    \item \textbf{EFAS Seasonal} --- European Flood Awareness System Seasonal forecasts
    \item \textbf{ESP} --- Ensemble Streamflow Prediction
    \item \textbf{ESA CCI} --- European Space Agency Climate Change Initiative (soil moisture)
    \item \textbf{ESSD} --- Earth System Science Data (journal)
    \item \textbf{FEST} --- Flash-flood Event-based Spatially-distributed Terrain-following (hydrological model)
    \item \textbf{GEUS} --- Geological Survey of Denmark and Greenland
    \item \textbf{GIA} --- Glacial Isostatic Adjustment
    \item \textbf{GMD} --- Geoscientific Model Development (journal)
    \item \textbf{GRACE FO} --- Gravity Recovery and Climate Experiment Follow-On
    \item \textbf{HIP} --- Hydrological Information Platform (Danish system)
    \item \textbf{HIPRAD} --- HIP Radar database
    \item \textbf{HPC Claaudia} --- High-Performance Computing facility at Aalborg University
    \item \textbf{ICON} --- ICOsahedral Nonhydrostatic model (DWD/ECMWF)
    \item \textbf{ICE-6G\_D} --- Glacial isostatic adjustment model (version 6G with denser temporal sampling)
    \item \textbf{LETKF} --- Local Ensemble Transform Kalman Filter
    \item \textbf{LSTM} --- Long Short-Term Memory (neural network architecture)
    \item \textbf{MCDI} --- Multi-pillar Combined Drought Index (new product)
    \item \textbf{MCMC} --- Markov Chain Monte Carlo
    \item \textbf{MODFLOW} --- Modular Three-Dimensional Finite-Difference Groundwater Flow Model
    \item \textbf{MSc} --- Master of Science
    \item \textbf{NeuralHydrology} --- Differentiable hydrology framework
    \item \textbf{PyGLDA} --- Python Geodesy and Land Data Assimilation (AAU research infrastructure)
    \item \textbf{ROC} --- Receiver Operating Characteristic
    \item \textbf{SEAS5} --- Seasonal Forecasting System 5 (ECMWF)
    \item \textbf{SMAP} --- Soil Moisture Active Passive (NASA satellite mission)
    \item \textbf{SPEI} --- Standardised Precipitation-Evapotranspiration Index
    \item \textbf{SPI} --- Standardised Precipitation Index
    \item \textbf{SRI} --- Standardised Runoff Index
    \item \textbf{SSI} --- Standardised Streamflow Index
    \item \textbf{SSMI} --- Standardised Soil Moisture Index
    \item \textbf{SGI} --- Standardised Groundwater Index
    \item \textbf{SIMGM} --- Simulating Groundwater Management (hydrological model)
    \item \textbf{STOTEN} --- Science of The Total Environment (journal)
    \item \textbf{USDM} --- United States Drought Monitor (severity classification)
    \item \textbf{VIC} --- Variable Infiltration Capacity (macroscale land surface model)
    \item \textbf{VICGlobal} --- Global parameter set for VIC
    \item \textbf{W3RA} --- Water Resources Reanalysis (Australian Bureau of Meteorology)
    \item \textbf{WGHM} --- WaterGAP Global Hydrology Model
    \item \textbf{WP1--WP5} --- Work Package 1 through 5
\end{itemize}
